void tft_init(tft_device_t *dev)
{
    dev->mutex = xSemaphoreCreateMutex();
    if (!dev->mutex) {
        ESP_LOGE(TAG, "Failed to create mutex");
        return;
    }

    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << dev->pin_dc) | (1ULL << dev->pin_rst) | (1ULL << dev->pin_cs),
        .mode = GPIO_MODE_OUTPUT,
        .pull_up_en = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type = GPIO_INTR_DISABLE
    };
    gpio_config(&io_conf);
    
    
    gpio_set_level(dev->pin_cs, 1);

    // аппаратный сброс
    gpio_set_level(dev->pin_rst, 0);
    vTaskDelay(pdMS_TO_TICKS(50));
    gpio_set_level(dev->pin_rst, 1);
    vTaskDelay(pdMS_TO_TICKS(50));



    spi_bus_config_t buscfg = {
        .miso_io_num = -1,
        .mosi_io_num = dev->pin_mosi,
        .sclk_io_num = dev->pin_clk,
        .quadwp_io_num = -1,
        .quadhd_io_num = -1,
        .max_transfer_sz = 4096
    };
    esp_err_t ret = spi_bus_initialize(SPI2_HOST, &buscfg, SPI_DMA_CH_AUTO);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "spi_bus_initialize failed: %d", ret);
        return;
    }

        spi_device_interface_config_t devcfg = {
        .clock_speed_hz = 26 * 1000 * 1000,
        .mode = 0,
        .spics_io_num = dev->pin_cs,
        .queue_size = 7,
        .flags = SPI_DEVICE_HALFDUPLEX
    };

    ret = spi_bus_add_device(SPI2_HOST, &devcfg, &dev->spi);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "spi_bus_add_device failed: %d", ret);
        return;
    }

    if (dev->spi == NULL) {
    ESP_LOGE(TAG, "SPI handle is NULL after successful add!");
    return;
    }

    ESP_LOGI(TAG, "SPI handle = %p", dev->spi);


    // Объединяем данные для каждой команды
    tft_send_cmd_with_data(dev,0x01, (uint8_t[]){0x00}, 1);
    vTaskDelay(pdMS_TO_TICKS(150));
    
    // Команда 0xCB с 5 байтами данных
    tft_send_cmd_with_data(dev,0xCB, (uint8_t[]){0x39, 0x2C, 0x00, 0x34, 0x02}, 5);
    
    // Команда 0xCF с 3 байтами данных
    tft_send_cmd_with_data(dev,0xCF, (uint8_t[]){0x00, 0xC1, 0x30}, 3);
    
    // Команда 0xE8 с 3 байтами данных
    tft_send_cmd_with_data(dev,0xE8, (uint8_t[]){0x85, 0x00, 0x78}, 3);
    
    // Команда 0xEA с 2 байтами данных
    tft_send_cmd_with_data(dev,0xEA, (uint8_t[]){0x00, 0x00}, 2);
    
    // Команда 0xED с 4 байтами данных
    tft_send_cmd_with_data(dev,0xED, (uint8_t[]){0x64, 0x03, 0x12, 0x81}, 4);
    
    // Команда 0xF7 с 1 байтом данных
    tft_send_cmd_with_data(dev,0xF7, (uint8_t[]){0x20}, 1);
    
    // Команда 0xC0 с 1 байтом данных
    tft_send_cmd_with_data(dev,0xC0, (uint8_t[]){0x23}, 1);
    
    // Команда 0xC1 с 1 байтом данных
    tft_send_cmd_with_data(dev,0xC1, (uint8_t[]){0x10}, 1);
    
    // Команда 0xC5 с 2 байтами данных
    tft_send_cmd_with_data(dev,0xC5, (uint8_t[]){0x3E, 0x28}, 2);
    
    // Команда 0xC7 с 1 байтом данных
    tft_send_cmd_with_data(dev,0xC7, (uint8_t[]){0x86}, 1);
    
    // Команда 0x36 с 1 байтом данных (ориентация)
    tft_send_cmd_with_data(dev,0x36, (uint8_t[]){0x48}, 1);
    
    // Команда 0x3A с 1 байтом данных (цветовой формат - 0x55 = 16 бит)
    tft_send_cmd_with_data(dev,0x3A, (uint8_t[]){0x55}, 1);
    
    // Команда 0xB1 с 2 байтами данных
    tft_send_cmd_with_data(dev,0xB1, (uint8_t[]){0x00, 0x18}, 2);
    
    // Команда 0xB6 с 3 байтами данных
    tft_send_cmd_with_data(dev,0xB6, (uint8_t[]){0x08, 0x82, 0x27}, 3);
    
    // Команда 0xF2 с 1 байтом данных
    tft_send_cmd_with_data(dev,0xF2, (uint8_t[]){0x00}, 1);
    
    // Команда 0x26 с 1 байтом данных
    tft_send_cmd_with_data(dev,0x26, (uint8_t[]){0x01}, 1);
    
    // Гамма-кривые (по 15 байт каждая)
    tft_send_cmd_with_data(dev,0xE0, (uint8_t[]){0x0F, 0x31, 0x2B, 0x0C, 0x0E, 
                                             0x08, 0x4E, 0xF1, 0x37, 0x07, 
                                             0x10, 0x03, 0x0E, 0x09, 0x00}, 15);
    
    tft_send_cmd_with_data(dev,0xE1, (uint8_t[]){0x00, 0x0E, 0x14, 0x03, 0x11, 
                                             0x07, 0x31, 0xC1, 0x48, 0x08, 
                                             0x0F, 0x0C, 0x31, 0x36, 0x0F}, 15);
    
    // Выход из сна
    tft_send_cmd_with_data(dev,0x11, NULL, 0);
    vTaskDelay(pdMS_TO_TICKS(120));
    
    // Включение дисплея
    tft_send_cmd_with_data(dev,0x29, NULL, 0);
}


void tft_send_cmd(tft_device_t *dev, uint8_t cmd)
{
    if (xSemaphoreTake(dev->mutex, portMAX_DELAY) != pdTRUE) {
        return;
    }
    
    gpio_set_level(dev->pin_dc, 0);
    spi_transaction_t t = {
        .length = 8,
        .tx_buffer = &cmd
    };
    spi_device_transmit(dev->spi, &t);
    
    xSemaphoreGive(dev->mutex);
    

        esp_err_t ret = spi_device_transmit(dev->spi, &t);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "spi_device_transmit failed: %s", esp_err_to_name(ret));
        xSemaphoreGive(dev->mutex);
        return;
    }
}

void tft_send_cmd_with_data(tft_device_t *dev, uint8_t cmd, const uint8_t *data, int len)
{
    if (xSemaphoreTake(dev->mutex, portMAX_DELAY) != pdTRUE) {
        return;
    }
    
    // Отправляем команду
    gpio_set_level(dev->pin_dc, 0);
    spi_transaction_t t_cmd = {
        .length = 8,
        .tx_buffer = &cmd
    };
    spi_device_transmit(dev->spi, &t_cmd);
    
    // Отправляем данные
    if (len > 0) {
        gpio_set_level(dev->pin_dc, 1);
        spi_transaction_t t_data = {
            .length = len * 8,
            .tx_buffer = data
        };
        spi_device_transmit(dev->spi, &t_data);
    }
    
    xSemaphoreGive(dev->mutex);
}

void tft_fill_pixel(tft_device_t *dev, int x, int y, uint8_t color1, uint8_t color2)
{
    
        // 1. Устанавливаем область на весь экран
        uint8_t col_data[] = {
        (x >> 8) & 0xFF,  // Старший байт X
        x & 0xFF,         // Младший байт X
        (x >> 8) & 0xFF,  // Старший байт X (конец)
        x & 0xFF          // Младший байт X (конец)
    };
        uint8_t row_data[] = {
        (y >> 8) & 0xFF,  // Старший байт y
        y & 0xFF,         // Младший байт y
        (y >> 8) & 0xFF,  // Старший байт y (конец)
        y & 0xFF          // Младший байт y (конец)
    };
    
  uint8_t color_data[] = {color1, color2};

    tft_send_cmd_with_data(dev, 0x2A, col_data, 4);
    tft_send_cmd_with_data(dev, 0x2B, row_data, 4);
    tft_send_cmd(dev, 0x2C);
    
    if (xSemaphoreTake(dev->mutex, portMAX_DELAY) == pdTRUE) {
    gpio_set_level(dev->pin_dc, 1);
    spi_transaction_t t = {
        .length = 16,
        .tx_buffer = color_data
    };
    spi_device_transmit(dev->spi, &t);
    
     xSemaphoreGive(dev->mutex);
    }

    //ESP_LOGI(TAG, "Screen filled with black: %d bytes sent", bytes_to_send);
}

void tft_fill_screen(tft_device_t *dev)
{
        // 1. Устанавливаем область на весь экран
    uint8_t col_data[] = {0x00, 0x00, 0x00, 0xEF}; // X: 0-319
    uint8_t row_data[] = {0x00, 0x00, 0x01, 0x3F}; // Y: 0-239
    
    tft_send_cmd_with_data(dev, 0x2A, col_data, 4);
    tft_send_cmd_with_data(dev, 0x2B, row_data, 4);
    tft_send_cmd(dev, 0x2C); // Memory write
    
    // 2. Создаем буфер с черным цветом
    #define BUF_SIZE 1024 // Отправляем по 1 КБ за раз
    
    // Важно: использовать DMA-совместимую память!
    uint8_t *buf = heap_caps_malloc(BUF_SIZE, MALLOC_CAP_DMA);
    if (!buf) {
        ESP_LOGE(TAG, "Failed to allocate buffer");
        return;
    }
    
    // Заполняем буфер нулями (черный цвет)
    
    //uint8_t color_data[] = {0xFF, 0xFF};

    memset(buf, 0, BUF_SIZE);
    
    
    // 3. Отправляем буфер много раз
    int bytes_to_send = 240 * 320 * 2; // Всего байт для отправки
    int remaining = bytes_to_send;
    
    while (remaining > 0) {
        int send_size = (remaining > BUF_SIZE) ? BUF_SIZE : remaining;
        
        // Отправляем кусок буфера
        if (xSemaphoreTake(dev->mutex, portMAX_DELAY) == pdTRUE) {
            gpio_set_level(dev->pin_dc, 1); // Режим данных
            
            spi_transaction_t t = {
                .length = send_size * 8,  // В битах
                .tx_buffer = buf
            };
            spi_device_transmit(dev->spi, &t);
            
            xSemaphoreGive(dev->mutex);
        }
        
        remaining -= send_size;
    }
    
    free(buf);
    //ESP_LOGI(TAG, "Screen filled with black: %d bytes sent", bytes_to_send);
}

    void letter_print(tft_device_t *dev, char letter,  uint8_t y_start, uint8_t x_start, uint8_t y_finish, uint8_t x_finish,  uint16_t color)
{
    uint8_t col_data[] = {x_start >> 8, x_start & 0xFF, x_finish >> 8, x_finish & 0xFF};
    uint8_t row_data[] = {y_start >> 8, y_start & 0xFF, y_finish >> 8, y_finish & 0xFF};

    tft_send_cmd_with_data(dev, 0x2A, col_data, 4); 
    tft_send_cmd_with_data(dev, 0x2B, row_data, 4);
    tft_send_cmd(dev, 0x2C);
